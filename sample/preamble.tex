% リスティングの設定
% \begin{lstlisting}[caption=foo, label=bar]　のように使う。
\usepackage{listings}
\lstset{
  basicstyle={\ttfamily},
  identifierstyle={\small},
  commentstyle={\smallitshape},
  keywordstyle={\small\bfseries},
  ndkeywordstyle={\small},
  stringstyle={\small\ttfamily},
  frame={tlRB},
  breaklines=true,
  columns=fixed,
  basewidth=0.5em,
  numberstyle={\scriptsize},
  stepnumber=1,
  numbersep=1zw,
  keepspaces=true,
  lineskip=-0.5ex
}

% 索引を作る
\usepackage{makeidx}

  setpagesize=false,
  bookmarksnumbered=true,%
  bookmarksopen=true,%
  colorlinks=true,%
  allcolors=blue
}

% 言葉を強調し、かつ索引に登録する
% \term{言葉}
% \term[よみがな]{言葉}
% のいずれかを使う。よみがなを指定した場合、検索にはよみがな順で
% 掲載される。よみがなは{}ではなく[]でくるむことに注意。

\newcommand*{\term}[2][]{%
  {\sffamily\bfseries #2}%
  \ifx\relax#1\relax\index{#2}\else\index{#1@#2}\fi }
